文件：
    open()功能的使用
    1、模式介绍
        控制文件读写内容的模式
            t文本
                1、读写都以str（unicode）为单位的
                2、只针对文本文件
                3、必须指定encoding='utf-8'
                4、t模式会将读出来的内容解码成unicode（更方便）
            b二进制/bytes
            	1、读写都是以bytes为单位
                2、可以针对所有文件
                3、一定不能指定字符编码
        控制文件读写操作的模式
            r只读模式
            w只写模式
            a只追加写模式
            +：r+、w+、a+
            
    2、基本操作流程
        打开文件
        读写文件
        关闭文件

    4、详细介绍文件模式

#以t模式为基础进行内存操作

#1、r（默认的操作模式）：只读模式 ,当文件不存在会报错,当文件存在时，文件指针跳到最开始位置
with open('file',mode='r+t',encoding='utf-8') as f:
    #res = f.read()      #读出硬盘的二进制>t模式将二进制换成字符unicode(解码的过程，需要指定一个原来的字符编码格式来解码)>再由unicode将其转换成字符
    f.write('1')



#2、w：只写模式，当文件不存在时会常见空文件，当文件存在会清空文件，指针位于开始位置
with open('d.txt',mode='wt',encoding='utf-8')as f:
    #f.read()    #不可读
    f.write('hahahaa\n')


#3、a：只追加写，在文件不存在时会创建空文档，当文件存在时指针直接跳到末尾
with open('e.txt',mode='at',encoding='utf-8')as f:
    #f.read()报错不会读
    f.write('猛')

指针移动的单位都是以bytes/字节为单位
只有一种情况特殊：
           t模式下read（n），n代表的是字符个数
           
f.seek(n,模式)：n指的是指针移动的字节个数
   模式0：参照物是文件开头位置
		f.seek(9,0)
		f.seek(3,0) #3
   模式1:参照物是当前指针所在位置
		f.seek(9,1)
		f.seek(3,1) #12
   模式2:参照物是文件末尾，应该倒着移动
		f.seek(9,2) #3
		f.seek(3,2) #9
强调：只有0模式可以在t下使用，1，2必须在b模式下用
		f.tell() #获取文件当前指针位置

#示范
with open('file',mode='rb')as f:
    f.seek(9,0)
    f.seek(3,0) #3
    print(f.tell())

with open(r'{}'.format(src_file),mode='rt',encoding='utf-8') as f1,\
    open(r'{}'.format(dis_file),mode='wt',encoding='utf-8') as f2:
    for line in f1:
        f2.write(f2)

#案例r模式实现登录
inp_username = input('your name:').strip()
inp_password = input('your password:').strip()

with open('user',mode='rt',encoding='utf-8')as f:
    for line in f:
        username,password = line.strip().split(':')
        if inp_username == username and inp_password == password:
            print('登录成功')
            break
    else:
        print('账号或密码错误')

with open(r'file',mode='rt',encoding='utf-8') as f:

        res=f.read(2)

        print(len(res),res)

#二：写相关操作
#f.writelines():  将列表中数据写入文件
with open('file',mode='wb')as f:
    l=[b'asd',b'ccc']
    f.writelines(l)
    #f.write('111\n222\n3333\n4444\n')
#补充1：如果纯英文字符，可以直接加前缀b得到bytes类型
l=[b'asd',b'ccc']

#补充2：'上'.encode('utf-8')等同于bytes（'上'，encoding='utf-8')
l = [bytes('一',encoding='utf-8'),bytes('二',encoding='utf-8')]
f.writelines()

#补充3：flush将内存中的数据刷到硬盘中
with open('file',mode='wb')as f:
    f.write('123')
    f.flush()

#f.seek的应用，实现tail -f查看日志功能
import time
with  open('file',mode='rb')as f:
    #1、将指针跳到文件末尾
    #f.read() #错误
    f.seek(0,2)
    while True:
        line = f.readline()
        if len(line) == 0:
            time.sleep(0.3)
        else:
            print(line.decode('utf-8'),end='')

#方式一:浪费内存空间，没浪费硬盘空间
with open('file',mode='rt',encoding='utf-8')as f:
    res=f.read()
    data=res.replace('abc','大宝贝')
    print(data)
with open('file1',mode='wt',encoding='utf-8')as f:
    f.write(data)


#方式二：浪费硬盘空间，没浪费内存空间
import os
with open('file',mode='rt',encoding='utf-8')as f,\
    open('file.swap',mode='wt',encoding='utf-8')as f1:
    for line in f:
        f1.write(line.replace('abc','宝贝大'))
os.remove('file')
os.rename('file.swap','file')
